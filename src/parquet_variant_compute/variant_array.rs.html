37/// Arrow Variant [`ExtensionType`].
38///
39/// Represents the canonical Arrow Extension Type for storing variants.
40/// See [`VariantArray`] for more examples of using this extension type.
41pub struct VariantType;
42
43impl ExtensionType for VariantType {
44    const NAME: &'static str = "arrow.parquet.variant";
45
46    // Variants extension metadata is an empty string
47    // <https://github.com/apache/arrow/blob/d803afcc43f5d132506318fd9e162d33b2c3d4cd/docs/source/format/CanonicalExtensions.rst?plain=1#L473>
48    type Metadata = &'static str;
49
50    fn metadata(&self) -> &Self::Metadata {
51        &""
52    }
53
54    fn serialize_metadata(&self) -> Option<String> {
55        Some(String::new())
56    }
57
58    fn deserialize_metadata(_metadata: Option<&str>) -> Result<Self::Metadata, ArrowError> {
59        Ok("")
60    }
61
62    fn supports_data_type(&self, data_type: &DataType) -> Result<(), ArrowError> {
63        if matches!(data_type, DataType::Struct(_)) {
64            Ok(())
65        } else {
66            Err(ArrowError::InvalidArgumentError(format!(
67                "VariantType only supports StructArray, got {data_type}"
68            )))
69        }
70    }
71
72    fn try_new(data_type: &DataType, _metadata: Self::Metadata) -> Result<Self, ArrowError> {
73        Self.supports_data_type(data_type)?;
74        Ok(Self)
75    }

78/// An array of Parquet [`Variant`] values
79///
80/// A [`VariantArray`] wraps an Arrow [`StructArray`] that stores the underlying
81/// `metadata` and `value` fields, and adds convenience methods to access
82/// the [`Variant`]s.
83///
84/// See [`VariantArrayBuilder`] for constructing `VariantArray` row by row.
85///
86/// See the examples below from converting between `VariantArray` and
87/// `StructArray`.
88///
89/// [`VariantArrayBuilder`]: crate::VariantArrayBuilder
90///
91/// # Documentation
92///
93/// At the time of this writing, Variant has been accepted as an official
94/// extension type but not been published to the [official list of extension
95/// types] on the Apache Arrow website. See the [Extension Type for Parquet
96/// Variant arrow] ticket for more details.
97///
98/// [Extension Type for Parquet Variant arrow]: https://github.com/apache/arrow/issues/46908
99/// [official list of extension types]: https://arrow.apache.org/docs/format/CanonicalExtensions.html
100///
101/// # Example: Check if a [`StructArray`] has the [`VariantType`] extension
102///
103/// Arrow Arrays only provide [`DataType`], but the extension type information
104/// is stored on a [`Field`]. Thus, you must have access to the [`Schema`] or
105/// [`Field`] to check for the extension type.
106///
107/// [`Schema`]: arrow_schema::Schema
108/// ```
109/// # use arrow::array::StructArray;
110/// # use arrow_schema::{Schema, Field, DataType};
111/// # use parquet_variant::Variant;
112/// # use parquet_variant_compute::{VariantArrayBuilder, VariantArray, VariantType};
113/// # fn get_variant_array() -> VariantArray {
114/// #   let mut builder = VariantArrayBuilder::new(10);
115/// #   builder.append_variant(Variant::from("such wow"));
116/// #   builder.build()
117/// # }
118/// # fn get_schema() -> Schema {
119/// #   Schema::new(vec![
120/// #     Field::new("id", DataType::Int32, false),
121/// #     get_variant_array().field("var"),
122/// #   ])
123/// # }
124/// let schema = get_schema();
125/// assert_eq!(schema.fields().len(), 2);
126/// // first field is not a Variant
127/// assert!(schema.field(0).try_extension_type::<VariantType>().is_err());
128/// // second field is a Variant
129/// assert!(schema.field(1).try_extension_type::<VariantType>().is_ok());
130/// ```
131///
132/// # Example: Constructing the correct [`Field`] for a [`VariantArray`]
133///
134/// You can construct the correct [`Field`] for a [`VariantArray`] using the
135/// [`VariantArray::field`] method.
136///
137/// ```
138/// # use arrow_schema::{Schema, Field, DataType};
139/// # use parquet_variant::Variant;
140/// # use parquet_variant_compute::{VariantArrayBuilder, VariantArray, VariantType};
141/// # fn get_variant_array() -> VariantArray {
142/// #   let mut builder = VariantArrayBuilder::new(10);
143/// #   builder.append_variant(Variant::from("such wow"));
144/// #   builder.build()
145/// # }
146/// let variant_array = get_variant_array();
147/// // First field is an integer id, second field is a variant
148/// let schema = Schema::new(vec![
149///   Field::new("id", DataType::Int32, false),
150///   // call VariantArray::field to get the correct Field
151///   variant_array.field("var"),
152/// ]);
153/// ```
154///
155/// You can also construct the [`Field`] using [`VariantType`] directly
156///
157/// ```
158/// # use arrow_schema::{Schema, Field, DataType};
159/// # use parquet_variant::Variant;
160/// # use parquet_variant_compute::{VariantArrayBuilder, VariantArray, VariantType};
161/// # fn get_variant_array() -> VariantArray {
162/// #   let mut builder = VariantArrayBuilder::new(10);
163/// #   builder.append_variant(Variant::from("such wow"));
164/// #   builder.build()
165/// # }
166/// # let variant_array = get_variant_array();
167/// // The DataType of a VariantArray varies depending on how it is shredded
168/// let data_type = variant_array.data_type().clone();
169/// // First field is an integer id, second field is a variant
170/// let schema = Schema::new(vec![
171///   Field::new("id", DataType::Int32, false),
172///   Field::new("var", data_type, false)
173///     // Add extension metadata to the field using `VariantType`
174///     .with_extension_type(VariantType),
175/// ]);
176/// ```
177///
178/// # Example: Converting a [`VariantArray`] to a [`StructArray`]
179///
180/// ```
181/// # use arrow::array::StructArray;
182/// # use parquet_variant::Variant;
183/// # use parquet_variant_compute::VariantArrayBuilder;
184/// // Create Variant Array
185/// let mut builder = VariantArrayBuilder::new(10);
186/// builder.append_variant(Variant::from("such wow"));
187/// let variant_array = builder.build();
188/// // convert to StructArray
189/// let struct_array: StructArray = variant_array.into();
190/// ```
191///
192/// # Example: Converting a [`StructArray`] to a [`VariantArray`]
193///
194/// ```
195/// # use arrow::array::StructArray;
196/// # use parquet_variant::Variant;
197/// # use parquet_variant_compute::{VariantArrayBuilder, VariantArray};
198/// # fn get_struct_array() -> StructArray {
199/// #   let mut builder = VariantArrayBuilder::new(10);
200/// #   builder.append_variant(Variant::from("such wow"));
201/// #   builder.build().into()
202/// # }
203/// let struct_array: StructArray = get_struct_array();
204/// // try and create a VariantArray from it
205/// let variant_array = VariantArray::try_new(&struct_array).unwrap();
206/// assert_eq!(variant_array.value(0), Variant::from("such wow"));
207/// ```
208///
209#[derive(Clone, Debug)]
210pub struct VariantArray {
211    /// Reference to the underlying StructArray
212    inner: StructArray,
213
214    /// The metadata column of this variant
215    metadata: BinaryViewArray,
216
217    /// how is this variant array shredded?
218    shredding_state: ShreddingState,
219}

221impl VariantArray {
222    /// Creates a new `VariantArray` from a [`StructArray`].
223    ///
224    /// # Arguments
225    /// - `inner` - The underlying [`StructArray`] that contains the variant data.
226    ///
227    /// # Returns
228    /// - A new instance of `VariantArray`.
229    ///
230    /// # Errors:
231    /// - If the `StructArray` does not contain the required fields
232    ///
233    /// # Requirements of the `StructArray`
234    ///
235    /// 1. A required field named `metadata` which is binary, large_binary, or
236    ///    binary_view
237    ///
238    /// 2. An optional field named `value` that is binary, large_binary, or
239    ///    binary_view
240    ///
241    /// 3. An optional field named `typed_value` which can be any primitive type
242    ///    or be a list, large_list, list_view or struct
243    ///
244    /// NOTE: It is also permissible for the metadata field to be
245    /// Dictionary-Encoded, preferably (but not required) with an index type of
246    /// int8.
247    ///
248    /// Currently, only [`BinaryViewArray`] are supported.
249    pub fn try_new(inner: &dyn Array) -> Result<Self, ArrowError> {
250        // Workaround lack of support for Binary
251        // https://github.com/apache/arrow-rs/issues/8387
252        let inner = cast_to_binary_view_arrays(inner)?;
253
254        let Some(inner) = inner.as_struct_opt() else {
255            return Err(ArrowError::InvalidArgumentError(
256                "Invalid VariantArray: requires StructArray as input".to_string(),
257            ));
258        };
259
260        // Note the specification allows for any order so we must search by name
261
262        // Ensure the StructArray has a metadata field of BinaryView
263        let Some(metadata_field) = inner.column_by_name("metadata") else {
264            return Err(ArrowError::InvalidArgumentError(
265                "Invalid VariantArray: StructArray must contain a 'metadata' field".to_string(),
266            ));
267        };
268        let Some(metadata) = metadata_field.as_binary_view_opt() else {
269            return Err(ArrowError::NotYetImplemented(format!(
270                "VariantArray 'metadata' field must be BinaryView, got {}",
271                metadata_field.data_type()
272            )));
273        };
274
275        // Extract value and typed_value fields
276        let value = if let Some(value_col) = inner.column_by_name("value") {
277            if let Some(binary_view) = value_col.as_binary_view_opt() {
278                Some(binary_view.clone())
279            } else {
280                return Err(ArrowError::NotYetImplemented(format!(
281                    "VariantArray 'value' field must be BinaryView, got {}",
282                    value_col.data_type()
283                )));
284            }
285        } else {
286            None
287        };
288        let typed_value = inner.column_by_name("typed_value").cloned();
289
290        // Note these clones are cheap, they just bump the ref count
291        Ok(Self {
292            inner: inner.clone(),
293            metadata: metadata.clone(),
294            shredding_state: ShreddingState::new(value, typed_value),
295        })
296    }
297
298    pub(crate) fn from_parts(
299        metadata: BinaryViewArray,
300        value: Option<BinaryViewArray>,
301        typed_value: Option<ArrayRef>,
302        nulls: Option<NullBuffer>,
303    ) -> Self {
304        let mut builder =
305            StructArrayBuilder::new().with_field("metadata", Arc::new(metadata.clone()), false);
306        if let Some(value) = value.clone() {
307            builder = builder.with_field("value", Arc::new(value), true);
308        }
309        if let Some(typed_value) = typed_value.clone() {
310            builder = builder.with_field("typed_value", typed_value, true);
311        }
312        if let Some(nulls) = nulls {
313            builder = builder.with_nulls(nulls);
314        }
315
316        Self {
317            inner: builder.build(),
318            metadata,
319            shredding_state: ShreddingState::new(value, typed_value),
320        }
321    }

334    pub fn shredding_state(&self) -> &ShreddingState {
335        &self.shredding_state

338    /// Return the [`Variant`] instance stored at the given row
339    ///
340    /// Note: This method does not check for nulls and the value is arbitrary
341    /// (but still well-defined) if [`is_null`](Self::is_null) returns true for the index.
342    ///
343    /// # Panics
344    /// * if the index is out of bounds
345    /// * if the array value is null
346    ///
347    /// If this is a shredded variant but has no value at the shredded location, it
348    /// will return [`Variant::Null`].
349    ///
350    ///
351    /// # Performance Note
352    ///
353    /// This is certainly not the most efficient way to access values in a
354    /// `VariantArray`, but it is useful for testing and debugging.
355    ///
356    /// Note: Does not do deep validation of the [`Variant`], so it is up to the
357    /// caller to ensure that the metadata and value were constructed correctly.
358    pub fn value(&self, index: usize) -> Variant<'_, '_> {
359        match (self.typed_value_field(), self.value_field()) {
360            // Always prefer typed_value, if available
361            (Some(typed_value), value) if typed_value.is_valid(index) => {
362                typed_value_to_variant(typed_value, value, index)
363            }
364            // Otherwise fall back to value, if available
365            (_, Some(value)) if value.is_valid(index) => {
366                Variant::new(self.metadata.value(index), value.value(index))
367            }
368            // It is technically invalid for neither value nor typed_value fields to be available,
369            // but the spec specifically requires readers to return Variant::Null in this case.
370            _ => Variant::Null,
371        }
372    }

451/// One shredded field of a partially or prefectly shredded variant. For example, suppose the
452/// shredding schema for variant `v` treats it as an object with a single field `a`, where `a` is
453/// itself a struct with the single field `b` of type INT. Then the physical layout of the column
454/// is:
455///
456/// ```text
457/// v: VARIANT {
458///     metadata: BINARY,
459///     value: BINARY,
460///     typed_value: STRUCT {
461///         a: SHREDDED_VARIANT_FIELD {
462///             value: BINARY,
463///             typed_value: STRUCT {
464///                 a: SHREDDED_VARIANT_FIELD {
465///                     value: BINARY,
466///                     typed_value: INT,
467///                 },
468///             },
469///         },
470///     },
471/// }
472/// ```
473///
474/// In the above, each row of `v.value` is either a variant value (shredding failed, `v` was not an
475/// object at all) or a variant object (partial shredding, `v` was an object but included unexpected
476/// fields other than `a`), or is NULL (perfect shredding, `v` was an object containing only the
477/// single expected field `a`).
478///
479/// A similar story unfolds for each `v.typed_value.a.value` -- a variant value if shredding failed
480/// (`v:a` was not an object at all), or a variant object (`v:a` was an object with unexpected
481/// additional fields), or NULL (`v:a` was an object containing only the single expected field `b`).
482///
483/// Finally, `v.typed_value.a.typed_value.b.value` is either NULL (`v:a.b` was an integer) or else a
484/// variant value (which could be `Variant::Null`).
485#[derive(Debug)]
486pub struct ShreddedVariantFieldArray {
487    /// Reference to the underlying StructArray
488    inner: StructArray,
489    shredding_state: ShreddingState,
490}

620/// Represents the shredding state of a [`VariantArray`]
621///
622/// [`VariantArray`]s can be shredded according to the [Parquet Variant
623/// Shredding Spec]. Shredding means that the actual value is stored in a typed
624/// `typed_field` instead of the generic `value` field.
625///
626/// Both value and typed_value are optional fields used together to encode a
627/// single value. Values in the two fields must be interpreted according to the
628/// following table (see [Parquet Variant Shredding Spec] for more details):
629///
630/// | value    | typed_value  | Meaning |
631/// |----------|--------------|---------|
632/// | NULL     | NULL         | The value is missing; only valid for shredded object fields |
633/// | non-NULL | NULL         | The value is present and may be any type, including [`Variant::Null`] |
634/// | NULL     | non-NULL     | The value is present and is the shredded type |
635/// | non-NULL | non-NULL     | The value is present and is a partially shredded object |
636///
637///
638/// Applying the above rules to entire columns, we obtain the following:
639///
640/// | value  | typed_value  | Meaning |
641/// |--------|-------------|---------|
642/// | --     | --          | **Missing**: The value is always missing; only valid for shredded object fields |
643/// | exists | --          | **Unshredded**: If present, the value may be any type, including [`Variant::Null`]
644/// | --     | exists      | **Perfectly shredded**: If present, the value is always the shredded type |
645/// | exists | exists      | **Imperfectly shredded**: The value might (not) be present and might (not) be the shredded type |
646///
647/// NOTE: Partial shredding is a row-wise situation that can arise under imperfect shredding (a
648/// column-wise situation): When both columns exist (imperfect shredding) and the typed_value column
649/// is a struct, then both columns can be non-NULL for the same row if value is a variant object
650/// (partial shredding).
651///
652/// [Parquet Variant Shredding Spec]: https://github.com/apache/parquet-format/blob/master/VariantShredding.md#value-shredding
653#[derive(Clone, Debug)]
654pub struct ShreddingState {
655    value: Option<BinaryViewArray>,
656    typed_value: Option<ArrayRef>,
657}
658
659impl ShreddingState {
660    /// Create a new `ShreddingState` from the given `value` and `typed_value` fields
661    ///
662    /// Note you can create a `ShreddingState` from a &[`StructArray`] using
663    /// `ShreddingState::from(&struct_array)`, for example:
664    ///
665    /// ```no_run
666    /// # use arrow::array::StructArray;
667    /// # use parquet_variant_compute::ShreddingState;
668    /// # fn get_struct_array() -> StructArray {
669    /// #   unimplemented!()
670    /// # }
671    /// let struct_array: StructArray = get_struct_array();
672    /// let shredding_state = ShreddingState::from(&struct_array);
673    /// ```
674    pub fn new(value: Option<BinaryViewArray>, typed_value: Option<ArrayRef>) -> Self {
675        Self { value, typed_value }
676    }
677
678    /// Return a reference to the value field, if present
679    pub fn value_field(&self) -> Option<&BinaryViewArray> {
680        self.value.as_ref()
681    }
682
683    /// Return a reference to the typed_value field, if present
684    pub fn typed_value_field(&self) -> Option<&ArrayRef> {
685        self.typed_value.as_ref()
686    }
687
688    /// Slice all the underlying arrays
689    pub fn slice(&self, offset: usize, length: usize) -> Self {
690        Self {
691            value: self.value.as_ref().map(|v| v.slice(offset, length)),
692            typed_value: self.typed_value.as_ref().map(|tv| tv.slice(offset, length)),
693        }
694    }
695}
696
697impl From<&StructArray> for ShreddingState {
698    fn from(inner_struct: &StructArray) -> Self {
699        let value = inner_struct
700            .column_by_name("value")
701            .and_then(|col| col.as_binary_view_opt().cloned());
702        let typed_value = inner_struct.column_by_name("typed_value").cloned();
703
704        ShreddingState::new(value, typed_value)
705    }
706}

708/// Builds struct arrays from component fields
709///
710/// TODO: move to arrow crate
711#[derive(Debug, Default, Clone)]
712pub(crate) struct StructArrayBuilder {
713    fields: Vec<FieldRef>,
714    arrays: Vec<ArrayRef>,
715    nulls: Option<NullBuffer>,
716}
717
718impl StructArrayBuilder {
719    pub fn new() -> Self {
720        Default::default()
721    }
722
723    /// Add an array to this struct array as a field with the specified name.
724    pub fn with_field(mut self, field_name: &str, array: ArrayRef, nullable: bool) -> Self {
725        let field = Field::new(field_name, array.data_type().clone(), nullable);
726        self.fields.push(Arc::new(field));
727        self.arrays.push(array);
728        self
729    }
730
731    /// Set the null buffer for this struct array.
732    pub fn with_nulls(mut self, nulls: NullBuffer) -> Self {
733        self.nulls = Some(nulls);
734        self
735    }
736
737    pub fn build(self) -> StructArray {
738        let Self {
739            fields,
740            arrays,
741            nulls,
742        } = self;
743        StructArray::new(Fields::from(fields), arrays, nulls)
744    }
745}

747/// returns the non-null element at index as a Variant
748fn typed_value_to_variant<'a>(
749    typed_value: &'a ArrayRef,
750    value: Option<&BinaryViewArray>,
751    index: usize,
752) -> Variant<'a, 'a> {
753    let data_type = typed_value.data_type();
754    if value.is_some_and(|v| !matches!(data_type, DataType::Struct(_)) && v.is_valid(index)) {
755        // Only a partially shredded struct is allowed to have values for both columns
756        panic!("Invalid variant, conflicting value and typed_value");
757    }
758    match data_type {
759        DataType::Boolean => {
760            let boolean_array = typed_value.as_boolean();
761            let value = boolean_array.value(index);
762            Variant::from(value)
763        }
764        DataType::Date32 => {
765            let array = typed_value.as_primitive::<Date32Type>();
766            let value = array.value(index);
767            let date = Date32Type::to_naive_date(value);
768            Variant::from(date)
769        }
770        // 16-byte FixedSizeBinary alway corresponds to a UUID; all other sizes are illegal.
771        DataType::FixedSizeBinary(16) => {
772            let array = typed_value.as_fixed_size_binary();
773            let value = array.value(index);
774            Uuid::from_slice(value).unwrap().into() // unwrap is safe: slice is always 16 bytes
775        }
776        DataType::BinaryView => {
777            let array = typed_value.as_binary_view();
778            let value = array.value(index);
779            Variant::from(value)
780        }
781        DataType::Utf8 => {
782            let array = typed_value.as_string::<i32>();
783            let value = array.value(index);
784            Variant::from(value)
785        }
786        DataType::Int8 => {
787            primitive_conversion_single_value!(Int8Type, typed_value, index)
788        }
789        DataType::Int16 => {
790            primitive_conversion_single_value!(Int16Type, typed_value, index)
791        }
792        DataType::Int32 => {
793            primitive_conversion_single_value!(Int32Type, typed_value, index)
794        }
795        DataType::Int64 => {
796            primitive_conversion_single_value!(Int64Type, typed_value, index)
797        }
798        DataType::Float16 => {
799            primitive_conversion_single_value!(Float16Type, typed_value, index)
800        }
801        DataType::Float32 => {
802            primitive_conversion_single_value!(Float32Type, typed_value, index)
803        }
804        DataType::Float64 => {
805            primitive_conversion_single_value!(Float64Type, typed_value, index)
806        }
807        DataType::Timestamp(TimeUnit::Microsecond, Some(_)) => {
808            generic_conversion_single_value!(
809                TimestampMicrosecondType,
810                as_primitive,
811                |v| DateTime::from_timestamp_micros(v).unwrap(),
812                typed_value,
813                index
814            )
815        }
816        DataType::Timestamp(TimeUnit::Microsecond, None) => {
817            generic_conversion_single_value!(
818                TimestampMicrosecondType,
819                as_primitive,
820                |v| DateTime::from_timestamp_micros(v).unwrap().naive_utc(),
821                typed_value,
822                index
823            )
824        }
825        DataType::Timestamp(TimeUnit::Nanosecond, Some(_)) => {
826            generic_conversion_single_value!(
827                TimestampNanosecondType,
828                as_primitive,
829                DateTime::from_timestamp_nanos,
830                typed_value,
831                index
832            )
833        }
834        DataType::Timestamp(TimeUnit::Nanosecond, None) => {
835            generic_conversion_single_value!(
836                TimestampNanosecondType,
837                as_primitive,
838                |v| DateTime::from_timestamp_nanos(v).naive_utc(),
839                typed_value,
840                index
841            )
842        }
843        // todo other types here (note this is very similar to cast_to_variant.rs)
844        // so it would be great to figure out how to share this code
845        _ => {
846            // We shouldn't panic in production code, but this is a
847            // placeholder until we implement more types
848            // https://github.com/apache/arrow-rs/issues/8091
849            debug_assert!(
850                false,
851                "Unsupported typed_value type: {}",
852                typed_value.data_type()
853            );
854            Variant::Null
855        }
856    }
857}

859/// Workaround for lack of direct support for BinaryArray
860/// <https://github.com/apache/arrow-rs/issues/8387>
861///
862/// The values are read as
863/// * `StructArray<metadata: Binary, value: Binary>`
864///
865/// but VariantArray needs them as
866/// * `StructArray<metadata: BinaryView, value: BinaryView>`
867///
868/// So cast them to get the right type.
869fn cast_to_binary_view_arrays(array: &dyn Array) -> Result<ArrayRef, ArrowError> {
870    let new_type = canonicalize_and_verify_data_type(array.data_type())?;
871    cast(array, new_type.as_ref())
872}

874/// Validates whether a given arrow decimal is a valid variant decimal
875///
876/// NOTE: By a strict reading of the "decimal table" in the [shredding spec], each decimal type
877/// should have a width-dependent lower bound on precision as well as an upper bound (i.e. Decimal16
878/// with precision 5 is invalid because Decimal4 "covers" it). But the variant shredding integration
879/// tests specifically expect such cases to succeed, so we only enforce the upper bound here.
880///
881/// [shredding spec]: https://github.com/apache/parquet-format/blob/master/VariantEncoding.md#encoding-types
882fn is_valid_variant_decimal(p: &u8, s: &i8, max_precision: u8) -> bool {
883    (1..=max_precision).contains(p) && (0..=*p as i8).contains(s)
884}

886/// Recursively visits a data type, ensuring that it only contains data types that can legally
887/// appear in a (possibly shredded) variant array. It also replaces Binary fields with BinaryView,
888/// since that's what comes back from the parquet reader and what the variant code expects to find.
889fn canonicalize_and_verify_data_type(
890    data_type: &DataType,
891) -> Result<Cow<'_, DataType>, ArrowError> {
892    use DataType::*;
893
894    // helper macros
895    macro_rules! fail {
896        () => {
897            return Err(ArrowError::InvalidArgumentError(format!(
898                "Illegal shredded value type: {data_type}"
899            )))
900        };
901    }
902    macro_rules! borrow {
903        () => {
904            Cow::Borrowed(data_type)
905        };
906    }
907
908    let new_data_type = match data_type {
909        // Primitive arrow types that have a direct variant counterpart are allowed
910        Null | Boolean => borrow!(),
911        Int8 | Int16 | Int32 | Int64 | Float32 | Float64 => borrow!(),
912
913        // Unsigned integers and half-float are not allowed
914        UInt8 | UInt16 | UInt32 | UInt64 | Float16 => fail!(),
915
916        // Most decimal types are allowed, with restrictions on precision and scale
917        Decimal32(p, s) if is_valid_variant_decimal(p, s, 9) => borrow!(),
918        Decimal64(p, s) if is_valid_variant_decimal(p, s, 18) => borrow!(),
919        Decimal128(p, s) if is_valid_variant_decimal(p, s, 38) => borrow!(),
920        Decimal32(..) | Decimal64(..) | Decimal128(..) | Decimal256(..) => fail!(),
921
922        // Only micro and nano timestamps are allowed
923        Timestamp(TimeUnit::Microsecond | TimeUnit::Nanosecond, _) => borrow!(),
924        Timestamp(TimeUnit::Millisecond | TimeUnit::Second, _) => fail!(),
925
926        // Only 32-bit dates and 64-bit microsecond time are allowed.
927        Date32 | Time64(TimeUnit::Microsecond) => borrow!(),
928        Date64 | Time32(_) | Time64(_) | Duration(_) | Interval(_) => fail!(),
929
930        // Binary and string are allowed. Force Binary to BinaryView because that's what the parquet
931        // reader returns and what the rest of the variant code expects.
932        Binary => Cow::Owned(DataType::BinaryView),
933        BinaryView | Utf8 => borrow!(),
934
935        // UUID maps to 16-byte fixed-size binary; no other width is allowed
936        FixedSizeBinary(16) => borrow!(),
937        FixedSizeBinary(_) | FixedSizeList(..) => fail!(),
938
939        // We can _possibly_ allow (some of) these some day?
940        LargeBinary | LargeUtf8 | Utf8View | ListView(_) | LargeList(_) | LargeListView(_) => {
941            fail!()
942        }
943
944        // Lists and struct are allowed, maps and unions are not
945        List(field) => match canonicalize_and_verify_field(field)? {
946            Cow::Borrowed(_) => borrow!(),
947            Cow::Owned(new_field) => Cow::Owned(DataType::List(new_field)),
948        },
949        // Struct is used by the internal layout, and can also represent a shredded variant object.
950        Struct(fields) => {
951            // Avoid allocation unless at least one field changes, to avoid unnecessary deep cloning
952            // of the data type. Even if some fields change, the others are shallow arc clones.
953            let mut new_fields = std::collections::HashMap::new();
954            for (i, field) in fields.iter().enumerate() {
955                if let Cow::Owned(new_field) = canonicalize_and_verify_field(field)? {
956                    new_fields.insert(i, new_field);
957                }
958            }
959
960            if new_fields.is_empty() {
961                borrow!()
962            } else {
963                let new_fields = fields
964                    .iter()
965                    .enumerate()
966                    .map(|(i, field)| new_fields.remove(&i).unwrap_or_else(|| field.clone()));
967                Cow::Owned(DataType::Struct(new_fields.collect()))
968            }
969        }
970        Map(..) | Union(..) => fail!(),
971
972        // We can _possibly_ support (some of) these some day?
973        Dictionary(..) | RunEndEncoded(..) => fail!(),
974    };
975    Ok(new_data_type)
976}
977
978fn canonicalize_and_verify_field(field: &Arc<Field>) -> Result<Cow<'_, Arc<Field>>, ArrowError> {
979    let Cow::Owned(new_data_type) = canonicalize_and_verify_data_type(field.data_type())? else {
980        return Ok(Cow::Borrowed(field));
981    };
982    let new_field = field.as_ref().clone().with_data_type(new_data_type);
983    Ok(Cow::Owned(Arc::new(new_field)))
984}
