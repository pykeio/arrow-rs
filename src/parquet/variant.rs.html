18//! ‚ö†Ô∏è Experimental Support for reading and writing [`Variant`]s to / from Parquet files ‚ö†Ô∏è
19//!
20//! This is a üöß Work In Progress
21//!
22//! Note: Requires the `variant_experimental` feature of the `parquet` crate to be enabled.
23//!
24//! # Features
25//! * [`Variant`] represents variant value, which can be an object, list, or primitive.
26//! * [`VariantBuilder`] for building `Variant` values.
27//! * [`VariantArray`] for representing a column of Variant values.
28//! * [`json_to_variant`] and [`variant_to_json`] for converting to/from JSON.
29//! * [`cast_to_variant()`] for casting other Arrow arrays to `VariantArray`.
30//! * [`VariantType`] Arrow ExtensionType for Parquet Variant logical type.
31//!   [`variant_get`] to extracting a value by path and functions to convert
32//!   between `Variant` and JSON.
33//!
34//! # Example: Writing a Parquet file with Variant column
35//! ```rust
36//! # use parquet::variant::{VariantArray, VariantType, VariantArrayBuilder, VariantBuilderExt};
37//! # use std::sync::Arc;
38//! # use arrow_array::{Array, ArrayRef, RecordBatch};
39//! # use arrow_schema::{DataType, Field, Schema};
40//! # use parquet::arrow::ArrowWriter;
41//! # fn main() -> Result<(), parquet::errors::ParquetError> {
42//!  // Use the VariantArrayBuilder to build a VariantArray
43//!  let mut builder = VariantArrayBuilder::new(3);
44//!  builder.new_object().with_field("name", "Alice").finish(); // row 1: {"name": "Alice"}
45//!  builder.append_value("such wow"); // row 2: "such wow" (a string)
46//!  let array = builder.build();
47//!
48//!  // Since VariantArray is an ExtensionType, it needs to be converted
49//!  // to an ArrayRef and Field with the appropriate metadata
50//!  // before it can be written to a Parquet file
51//!  let field = array.field("data");
52//!  let array = ArrayRef::from(array);
53//!  // create a RecordBatch with the VariantArray
54//!  let schema = Schema::new(vec![field]);
55//!  let batch = RecordBatch::try_new(Arc::new(schema), vec![array])?;
56//!
57//!  // Now you can write the RecordBatch to the Parquet file, as normal
58//!  let file = std::fs::File::create("variant.parquet")?;
59//!  let mut writer = ArrowWriter::try_new(file, batch.schema(), None)?;
60//!  writer.write(&batch)?;
61//!  writer.close()?;
62//!
63//! # std::fs::remove_file("variant.parquet")?;
64//! # Ok(())
65//! # }
66//! ```
67//!
68//! # Example: Writing JSON into a Parquet file with Variant column
69//! ```rust
70//! # use std::sync::Arc;
71//! # use arrow_array::{ArrayRef, RecordBatch, StringArray};
72//! # use arrow_schema::Schema;
73//! # use parquet::variant::{json_to_variant, VariantArray};
74//! # use parquet::arrow::ArrowWriter;
75//! # fn main() -> Result<(), parquet::errors::ParquetError> {
76//!  // Create an array of JSON strings, simulating a column of JSON data
77//!  let input_array: ArrayRef = Arc::new(StringArray::from(vec![
78//!   Some(r#"{"name": "Alice", "age": 30}"#),
79//!   Some(r#"{"name": "Bob", "age": 25, "address": {"city": "New York"}}"#),
80//!   None,
81//!   Some("{}"),
82//!  ]));
83//!
84//!  // Convert the JSON strings to a VariantArray
85//!  let array: VariantArray = json_to_variant(&input_array)?;
86//!  // create a RecordBatch with the VariantArray
87//!  let schema = Schema::new(vec![array.field("data")]);
88//!  let batch = RecordBatch::try_new(Arc::new(schema), vec![ArrayRef::from(array)])?;
89//!
90//!  // write the RecordBatch to a Parquet file as normal
91//!  let file = std::fs::File::create("variant-json.parquet")?;
92//!  let mut writer = ArrowWriter::try_new(file, batch.schema(), None)?;
93//!  writer.write(&batch)?;
94//!  writer.close()?;
95//! # std::fs::remove_file("variant-json.parquet")?;
96//! # Ok(())
97//! # }
98//! ```
99//!
100//! # Example: Reading a Parquet file with Variant column
101//!
102//! Use the [`VariantType`] extension type to find the Variant column:
103//!
104//! ```
105//! # use std::sync::Arc;
106//! # use std::path::PathBuf;
107//! # use arrow_array::{ArrayRef, RecordBatch, RecordBatchReader};
108//! # use parquet::variant::{Variant, VariantArray, VariantType};
109//! # use parquet::arrow::arrow_reader::ArrowReaderBuilder;
110//! # fn main() -> Result<(), parquet::errors::ParquetError> {
111//! # use arrow_array::StructArray;
112//! # fn file_path() -> PathBuf { // return a testing file path
113//! #    PathBuf::from(arrow::util::test_util::parquet_test_data())
114//! #   .join("..")
115//! #   .join("shredded_variant")
116//! #   .join("case-075.parquet")
117//! # }
118//! // Read the Parquet file using standard Arrow Parquet reader.
119//! // Note this file has 2 columns: "id", "var", and the "var" column
120//  // contains a variant that looks like this:
121//  // "Variant(metadata=VariantMetadata(dict={}), value=Variant(type=STRING, value=iceberg))"
122//! let file = std::fs::File::open(file_path())?;
123//! let mut reader = ArrowReaderBuilder::try_new(file)?.build()?;
124//!
125//! // You can check if a column contains a Variant using
126//! // the VariantType extension type
127//! let schema = reader.schema();
128//! let field = schema.field_with_name("var")?;
129//! assert!(field.try_extension_type::<VariantType>().is_ok());
130//!
131//! // The reader will yield RecordBatches with a StructArray
132//! // to convert them to VariantArray, use VariantArray::try_new
133//! let batch = reader.next().unwrap().unwrap();
134//!
135//! let col = batch.column_by_name("var").unwrap();
136//! let var_array = VariantArray::try_new(col)?;
137//! assert_eq!(var_array.len(), 1);
138//! let var_value: Variant = var_array.value(0);
139//! assert_eq!(var_value, Variant::from("iceberg")); // the value in case-075.parquet
140//! # Ok(())
141//! # }
142//! ```

166    /// Ensure a file with Variant LogicalType, written by another writer in
167    /// parquet-testing, can be read as a VariantArray
168    #[test]
169    fn read_logical_type() {
170        // Note: case-075 2 columns ("id", "var")
171        // The variant looks like this:
172        // "Variant(metadata=VariantMetadata(dict={}), value=Variant(type=STRING, value=iceberg))"
173        let batch = read_shredded_variant_test_case("case-075.parquet");
174
175        assert_variant_metadata(&batch, "var");
176        let var_column = batch.column_by_name("var").expect("expected var column");
177        let var_array =
178            VariantArray::try_new(&var_column).expect("expected var column to be a VariantArray");
179
180        // verify the value
181        assert_eq!(var_array.len(), 1);
182        assert!(var_array.is_valid(0));
183        let var_value = var_array.value(0);
184        assert_eq!(var_value, Variant::from("iceberg"));
185    }

187    /// Writes a variant to a parquet file and ensures the parquet logical type
188    /// annotation is correct
189    #[test]
190    fn write_logical_type() {
191        let array = variant_array();
192        let batch = variant_array_to_batch(array);
193        let buffer = write_to_buffer(&batch);
194
195        // read the parquet file's metadata and verify the logical type
196        let metadata = read_metadata(&Bytes::from(buffer));
197        let schema = metadata.file_metadata().schema_descr();
198        let fields = schema.root_schema().get_fields();
199        assert_eq!(fields.len(), 1);
200        let field = &fields[0];
201        assert_eq!(field.name(), "data");
202        // data should have been written with the Variant logical type
203        assert_eq!(
204            field.get_basic_info().logical_type(),
205            Some(crate::basic::LogicalType::Variant)
206        );
207    }

209    /// Return a VariantArray with 3 rows:
210    ///
211    /// 1. `{"name": "Alice"}`
212    /// 2. `"such wow"` (a string)
213    /// 3. `null`
214    fn variant_array() -> VariantArray {
215        let mut builder = VariantArrayBuilder::new(3);
216        // row 1: {"name": "Alice"}
217        builder.new_object().with_field("name", "Alice").finish();
218        // row 2: "such wow" (a string)
219        builder.append_value("such wow");
220        // row 3: null
221        builder.append_null();
222        builder.build()
223    }

225    /// Writes a VariantArray to a parquet file and reads it back, verifying that
226    /// the data is the same
227    fn roundtrip(array: VariantArray) {
228        let source_batch = variant_array_to_batch(array);
229        assert_variant_metadata(&source_batch, "data");
230
231        let buffer = write_to_buffer(&source_batch);
232        let result_batch = read_to_batch(Bytes::from(buffer));
233        assert_variant_metadata(&result_batch, "data");
234        assert_eq!(result_batch, source_batch); // NB this also checks the schemas
235    }

237    /// creates a RecordBatch with a single column "data" from a VariantArray,
238    fn variant_array_to_batch(array: VariantArray) -> RecordBatch {
239        let field = array.field("data");
240        let schema = Schema::new(vec![field]);
241        RecordBatch::try_new(Arc::new(schema), vec![ArrayRef::from(array)]).unwrap()
242    }
243
244    /// writes a RecordBatch to memory buffer and returns the buffer
245    fn write_to_buffer(batch: &RecordBatch) -> Vec<u8> {
246        let mut buffer = vec![];
247        let mut writer = ArrowWriter::try_new(&mut buffer, batch.schema(), None).unwrap();
248        writer.write(batch).unwrap();
249        writer.close().unwrap();
250        buffer
251    }

253    /// Reads the Parquet metadata
254    fn read_metadata<T: ChunkReader + 'static>(input: &T) -> ParquetMetaData {
255        let mut reader = ParquetMetaDataReader::new();
256        reader.try_parse(input).unwrap();
257        reader.finish().unwrap()
258    }
259
260    /// Reads a RecordBatch from a reader (e.g. Vec or File)
261    fn read_to_batch<T: ChunkReader + 'static>(reader: T) -> RecordBatch {
262        let reader = ArrowReaderBuilder::try_new(reader)
263            .unwrap()
264            .build()
265            .unwrap();
266        let mut batches: Vec<RecordBatch> = reader.collect::<Result<Vec<_>, _>>().unwrap();
267        assert_eq!(batches.len(), 1);
268        batches.swap_remove(0)
269    }

271    /// Verifies the variant metadata is present in the schema for the specified
272    /// field name.
273    fn assert_variant_metadata(batch: &RecordBatch, field_name: &str) {
274        let schema = batch.schema();
275        let field = schema
276            .field_with_name(field_name)
277            .expect("could not find expected field");
278
279        // explicitly check the metadata so it is clear in the tests what the
280        // names are
281        let metadata_value = field
282            .metadata()
283            .get("ARROW:extension:name")
284            .expect("metadata does not exist");
285
286        assert_eq!(metadata_value, "arrow.parquet.variant");
287
288        // verify that `VariantType` also correctly finds the metadata
289        field
290            .try_extension_type::<VariantType>()
291            .expect("VariantExtensionType should be readable");
292    }
