18//! Geospatial statistics for Parquet files.
19//!
20//! This module provides functionality for working with geospatial statistics in Parquet files.
21//! It includes support for bounding boxes and geospatial statistics in column chunk metadata.
22
23use crate::format::GeospatialStatistics as TGeospatialStatistics;
24use crate::geospatial::bounding_box::BoundingBox;
25
26// ----------------------------------------------------------------------
27// Geospatial Statistics
28
29/// Represents geospatial statistics for a Parquet column or dataset.
30///
31/// This struct contains metadata about the spatial characteristics of geospatial data,
32/// including bounding box information and the types of geospatial geometries present.
33/// It's used to optimize spatial queries and provide spatial context for data analysis.
34///
35/// # Examples
36///
37/// ```
38/// use parquet::geospatial::statistics::GeospatialStatistics;
39/// use parquet::geospatial::bounding_box::BoundingBox;
40///
41/// // Statistics with bounding box
42/// let bbox = BoundingBox::new(0.0, 0.0, 100.0, 100.0);
43/// let stats = GeospatialStatistics::new(Some(bbox), Some(vec![1, 2, 3]));
44/// ```
45#[derive(Clone, Debug, PartialEq, Default)]
46pub struct GeospatialStatistics {
47    /// Optional bounding defining the spatial extent, where None represents a lack of information.
48    bbox: Option<BoundingBox>,
49    /// Optional list of geometry type identifiers, where None represents lack of information
50    geospatial_types: Option<Vec<i32>>,
51}
52
53impl GeospatialStatistics {
54    /// Creates a new geospatial statistics instance with the specified data.
55    pub fn new(bbox: Option<BoundingBox>, geospatial_types: Option<Vec<i32>>) -> Self {
56        Self {
57            bbox,
58            geospatial_types,
59        }
60    }
61}
62
63/// Converts a Thrift-generated geospatial statistics object to the internal representation.
64pub fn from_thrift(geo_statistics: Option<TGeospatialStatistics>) -> Option<GeospatialStatistics> {
65    let geo_stats = geo_statistics?;
66    let bbox = geo_stats.bbox.map(|bbox| bbox.into());
67    // If vector is empty, then set it to None
68    let geospatial_types: Option<Vec<i32>> = geo_stats.geospatial_types.filter(|v| !v.is_empty());
69    Some(GeospatialStatistics::new(bbox, geospatial_types))
70}
71
72/// Converts our internal geospatial statistics to the Thrift-generated format.
73pub fn to_thrift(geo_statistics: Option<&GeospatialStatistics>) -> Option<TGeospatialStatistics> {
74    let geo_stats = geo_statistics?;
75    let bbox = geo_stats.bbox.clone().map(|bbox| bbox.into());
76    let geospatial_types = geo_stats.geospatial_types.clone();
77    Some(TGeospatialStatistics::new(bbox, geospatial_types))
78}

84    /// Tests the conversion from Thrift format when no statistics are provided.
85    #[test]
86    fn test_from_thrift() {
87        assert_eq!(from_thrift(None), None);
88        assert_eq!(
89            from_thrift(Some(TGeospatialStatistics::new(None, None))),
90            Some(GeospatialStatistics::default())
91        );
92    }

94    /// Tests the conversion from Thrift format with actual geospatial data.
95    #[test]
96    fn test_geo_statistics_from_thrift() {
97        let bbox = BoundingBox::new(0.0, 0.0, 100.0, 100.0);
98        let geospatial_types = vec![1, 2, 3];
99        let stats = GeospatialStatistics::new(Some(bbox), Some(geospatial_types));
100        let thrift_stats = to_thrift(Some(&stats));
101        assert_eq!(from_thrift(thrift_stats), Some(stats));
102    }

104    #[test]
105    fn test_bbox_to_thrift() {
106        use crate::format as parquet;
107        use thrift::OrderedFloat;
108
109        let bbox = BoundingBox::new(0.0, 0.0, 100.0, 100.0);
110        let thrift_bbox: parquet::BoundingBox = bbox.into();
111        assert_eq!(thrift_bbox.xmin, 0.0);
112        assert_eq!(thrift_bbox.xmax, 0.0);
113        assert_eq!(thrift_bbox.ymin, 100.0);
114        assert_eq!(thrift_bbox.ymax, 100.0);
115        assert_eq!(thrift_bbox.zmin, None);
116        assert_eq!(thrift_bbox.zmax, None);
117        assert_eq!(thrift_bbox.mmin, None);
118        assert_eq!(thrift_bbox.mmax, None);
119
120        let bbox_z = BoundingBox::new(0.0, 0.0, 100.0, 100.0).with_zrange(5.0, 15.0);
121        let thrift_bbox_z: parquet::BoundingBox = bbox_z.into();
122        assert_eq!(thrift_bbox_z.zmin, Some(OrderedFloat(5.0)));
123        assert_eq!(thrift_bbox_z.zmax, Some(OrderedFloat(15.0)));
124
125        let bbox_m = BoundingBox::new(0.0, 0.0, 100.0, 100.0).with_mrange(10.0, 20.0);
126        let thrift_bbox_m: parquet::BoundingBox = bbox_m.into();
127        assert_eq!(thrift_bbox_m.mmin, Some(OrderedFloat(10.0)));
128        assert_eq!(thrift_bbox_m.mmax, Some(OrderedFloat(20.0)));
129    }

131    #[test]
132    fn test_read_geospatial_statistics_from_file() {
133        use crate::file::reader::{FileReader, SerializedFileReader};
134        use std::fs::File;
135
136        let path = format!(
137            "{}/geospatial/geospatial.parquet",
138            arrow::util::test_util::parquet_test_data(),
139        );
140        let file = File::open(path).unwrap();
141        let reader = SerializedFileReader::try_from(file).unwrap();
142        let metadata = reader.metadata();
143
144        // geospatial.parquet schema:
145        //    optional binary field_id=-1 group (String);
146        //    optional binary field_id=-1 wkt (String);
147        //    optional binary field_id=-1 geometry (Geometry(crs=));
148        let geo_statistics = metadata.row_group(0).column(2).geo_statistics();
149        assert!(geo_statistics.is_some());
150
151        let expected_bbox = BoundingBox::new(10.0, 40.0, 10.0, 40.0)
152            .with_zrange(30.0, 80.0)
153            .with_mrange(200.0, 1600.0);
154        let expected_geospatial_types = vec![
155            1, 2, 3, 4, 5, 6, 7, 1001, 1002, 1003, 1004, 1005, 1006, 1007, 2001, 2002, 2003, 2004,
156            2005, 2006, 2007, 3001, 3002, 3003, 3004, 3005, 3006, 3007,
157        ];
158        assert_eq!(
159            geo_statistics.unwrap().geospatial_types,
160            Some(expected_geospatial_types)
161        );
162        assert_eq!(geo_statistics.unwrap().bbox, Some(expected_bbox));
163    }
